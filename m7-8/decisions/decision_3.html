      const form = document.getElementById('form');
      const table = document.getElementById('table');
      
      form.addEventListener('submit', onSubmit);
      
      function onSubmit(evt) {
        evt.preventDefault();
        const {
          row: { value: rowVal },
          cell: { value: cellVal },
          text: { value: textVal },
        } = form.elements;
      
        table.rows[Number(rowVal)].cells[Number(cellVal)].textContent = textVal;
      }
