    const refs = {
        incBtn: document.querySelector('[data-action="increment"]'),
        decBtn: document.querySelector('[data-action="decrement"]'),
        stopBtn: document.querySelector('[data-action="stop"]'),
        text: document.querySelector('span'),
    };
    let timerId = null;
    
    refs.incBtn.addEventListener('click', () => updateTime(1));
    refs.decBtn.addEventListener('click', () => updateTime(-1));
    refs.stopBtn.addEventListener('click', () => clearTimeout(timerId));
    
    function updateTime(val) {
        refs.text.innerHTML = +refs.text.innerHTML + val;
        timerId = setTimeout(() => updateTime(val), 1000);
    }
