         const TAX = 5;
         let sallary = prompt('Enter the amount of your salary in a month.', 3500);
         let countOfMonth = +prompt('Enter the count of month.', 10);
         let totalEarned = 0;
         let totalAfterTax = 0;
         
         for (let i = countOfMonth; i > 0; i--) {
             const taxAmount = sallary / 100 * 5
             totalEarned += sallary - taxAmount;
             totalAfterTax += taxAmount;
         }
         
         console.log(`
         You earned ${totalEarned} within ${countOfMonth} months, after-tax ${TAX}%
         The total ammount of tax is ${totalAfterTax}.
         `);
